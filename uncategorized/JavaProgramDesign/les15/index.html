try catch finally处理异常
import java.io.PrintStream;
import java.util.InputMismatchException;
import java.util.Scanner;
/*
//2. 异常处理
//如果对于一个可能发生异常的语句，会产生多种类型的异常

//如果多种类型异常，处理方式一样
//则catch(Exception e){}
//写一个catch(){}就行

//如果多种类型异常，处理方式不一样
//则catch(){}catch(){}catch(){}……
//有多少种就得写多少个catch

//finally{}的代码无论是否处理异常，都一定会执行
//所以finally一般会用来就行资源释放（例如：文件的关闭，数据的保存）
try {
//可能发生异常的语句
}catch (**Exception){
//异常处理
}
*/
public class test {
public static void main(String[] args) {
Scanner sc=new Scanner(System.in);
while(true){
try{
int a=sc.nextInt();//error:write abc
int b=19/a;//error:let a=0
int c[]=new int[a];//error:let a=-4
c[0]=1;
c[1]=2;
c[2]=3;
PrintStream ps=new PrintStream(System.out,true,”utf-9”);
System.out.println(“Success！”);
}catch (InputMismatchException e){
System.out.println(“输入数据必须为整数”);
sc.next();
}catch (ArithmeticException e){
System.out.println(“输入数据不能为0”);
}catch (NegativeArraySizeException e){
System.out.println(“输入数据不能是负数”);
}catch (ArrayIndexOutOfBoundsException e){
System.out.println(“输入数据必须大于等于3”);
}catch (Exception e){//未知异常处理
System.out.println(“未知错误”);
}finally {
//【书P121】此代码块内语句及后续语句一定会被执行
System.out.println(“Always keep running code in this area.”);
//现实应用：finally代码块可以用来做释放资源
}
//catch是顺序执行的，不可以把父类异常Exception放在前面！！！
}
}
}

import java.util.Scanner;

/*
3.throw和throws关键字区别
3.1 throws关键字：函数使用throws关键字，将该函数中产生异常的语句抛出给使用该函数的代码去解决
如果有多个函数使用throws抛出异常，可以一起进行try catch统一处理
3.2 throw关键字：自定义异常类
*/
public class test {
public static void main(String[] args){
Scanner sc=new Scanner(System.in);
try{
int age= sc.nextInt();
MyException1 me1=new MyException1(age);
throw me1;
}catch (MyException1 me){
System.out.println(me.errorMessage);
}catch (Exception e){
System.out.println(“输入的不是整数数值”);
}
}
static void f1() throws ArithmeticException{
int a=3/0;
}
static void f2() throws NegativeArraySizeException{
int b[]=new int[-3];
}
}
