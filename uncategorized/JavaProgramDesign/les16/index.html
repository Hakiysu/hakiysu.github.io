集合
List集合的元素是有序的，可以重复出现
ArrayList：动态数组数据结构
LinkedList：链表数据结构
ArrayList访问元素速度优于LinkedList，LinkedList占用内存空间比较大，但LinkedList在批量插入or删除优于ArrayList
List集合关系元素是否有序，而不关心是否重复
import java.util.ArrayList;
import java.util.Iterator;
//1. 集合List接口–ArrayList LinkedList实现类
public class test {
public static void main(String[] args){
//1.1 ArrayList实例化
ArrayList a1=new ArrayList();
//1.2 增加元素
a1.add(“hello”);
a1.add(“world”);
a1.add(“a”);
a1.add(“b”);
a1.add(“c”);
a1.add(“d”);
System.out.println(a1);
//1.2.extra 指定位置之前增加
a1.add(1,”the”);
System.out.println(a1);

    //1.3 remove
    a1.remove(3);//删除指定位置元素
    System.out.println(a1);
    a1.remove("b");//删除完全匹配此字符串的元素
    System.out.println(a1);
    a1.remove("e");//删除的元素不存在此数组中，则不删除也不报错
    System.out.println(a1);

    //1.4 查找
    System.out.println(a1.get(3));//查找a1数组中3号位置的元素，等同于a1[3]
    //System.out.println(a1.get(9));//数组下标越界，导致异常

    //1.5 修改
    a1.set(1,"Hello");//将指定位置的元素修改为指定内容
    System.out.println(a1);

    //1.6 size()求数组长度
    //    isEmpty()判断数组是否为空
    //    clear()数组清空
    //    contain()判断是否包含某元素
    System.out.println(a1.size());
    System.out.println(a1.isEmpty());
    System.out.println(a1.contains("Hello"));
    
    //1.7 数组遍历
    //1.7.1 for循环
    for(int i=0;i< a1.size();i++)
        System.out.println(a1.get(i)+" ");
    //1.7.2 for-each遍历
    for(Object o:a1){
        System.out.println(o);
    }
    //1.7.3 迭代器的遍历
    Iterator it=a1.iterator();//从List中获取一个迭代器对象
    while (it.hasNext()){
        //如果有下一个元素，则it.hasNext()==true；否则为false
        System.out.println(it.next()+" ");
    }
}

}

Stack栈
import java.util.Stack;
//2. 栈和队列
public class test {
public static void main(String[] args){
//2.1 Stack类：栈 先进后出
Stack s=new Stack();
//2.2 进入栈
s.push(“1”);
s.push(“2”);
s.push(“3”);
s.push(“4”);
s.push(“5”);
System.out.println(s);
//2.3 出栈
Object o=s.pop();
System.out.println(o);
System.out.println(s);
//2.4 获取栈顶元素，不删除
Object o1=s.peek();
System.out.println(o1);
System.out.println(s);
}
}

Queue队列
import java.util.LinkedList;
import java.util.Queue;
//2. 栈和队列
public class test {
public static void main(String[] args) {
//Queue队列接口
Queue q=new LinkedList();
//2.1 入队列–offer()
q.offer(1);
q.offer(2);
q.offer(3);
q.offer(4);
System.out.println(q);
//2.2 出队列–获取队头节点，并且把队头节点删除–poll
Object o=q.poll();
System.out.println(o);
System.out.println(q);
//2.3 获取队头节点信息，不删除–peek
Object o1=q.peek();
System.out.println(o1);
System.out.println(q);
}
}

Map集合
键(key)集合Set类型：不能有重复元素
值(value)集合Collection类型：可以有重复元素
import java.util.HashMap;
//4. Map集合 键值对接口 HashMap
public class test {
public static void main(String[] args) {
//4.1 Map集合实例化
HashMap hm=new HashMap();
//4.2 增加键值对
hm.put(“CN”,”中国”);
hm.put(“cn”,”中国”);
hm.put(“China”,”中国”);
hm.put(“Russia”,”俄罗斯”);
System.out.println(hm);
//4.3 删除
hm.remove(“cn”);//根据键删除键值对
System.out.println(hm);
//4.4 根据键查找值
System.out.println(hm.get(“China”));
//4.5 遍历
Set s=hm.keySet();
Collection c=hm.values();
System.out.println(s);
System.out.println(c);
}
}
