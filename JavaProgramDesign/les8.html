以模拟🐕的信息录入的系统功能
import java.io.PrintStream;
import java.io.UnsupportedEncodingException;
import java.util.Scanner;

//一直录入dog信息，直到用户选择退出系统为止
//menu
//1.录入 2.展示所有🐕的信息 3.退出
//choose1：user输入dog的品种，年龄，姓名
//choose2：展示当前储存的所有dog信息（save in array）—数组类型 ？？？ 狗类型
//每一个数组元素都是dog类型–对象数组
//类名 数组名[]=new 类名[]’
//choose3：exit system
public class test {
public static void main(String [] args) throws UnsupportedEncodingException {
Scanner sc=new Scanner(System.in,”utf-8”);
PrintStream ps=new PrintStream(System.out,true,”utf-8”);
ps.println(“Menu:1.WriteData 2.ShowAllData 3.Exit”);
int choice=sc.nextInt();
if(choice==1){
ps.println(“Name:”);
String name=sc.next();
ps.println(“Category:”);
String category=sc.next();
ps.println(“Age:”);
int age=sc.nextInt();
ManagerSystem.writeData(category,age,name);
}
else if(choice==2){
ManagerSystem.printAllData();
}
else if(choice==3){
ManagerSystem.exitSystem();
}
}
}
class ManagerSystem{
static Dog d[]=new Dog[1024];//成员变量—dog数组，用于保存当前用于录入dog的信息
static int dogNumber=0;//save number of dog data
static void writeData(String category,int age,String name){//录入信息方法本质—将用户输入的信息存入到数组中
Dog tempDog=new Dog(category,age,name);//调用一次write，将实例化一只tempDog
d[dogNumber]=tempDog;//将tempDog保存到Dog数组中
dogNumber++;//狗的数量++
}
static void printAllData() throws UnsupportedEncodingException {
PrintStream ps=new PrintStream(System.out,true,”utf-8”);
if(dogNumber==0){
ps.println(“NO DOG DATA!!!”);
}
ps.println(“当前已录入”+dogNumber+”只🐕”);
for (int i=0;i<dogNumber;i++){
ps.println(“[“+dogNumber+”]：”+d[i].name+”-“+d[i].category+”-“+d[i].age);
}
}
static void exitSystem(){
//just break it
}
}
class Dog{
//定义dog成员变量
String category;
int age;
String name;
//构造函数对成员变量赋值
Dog(String category,int age,String name)
{
this.category=category;
this.age=age;
this.name=name;
}
}
