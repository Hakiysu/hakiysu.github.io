访问控制

//FILE A.java
package pak1;
//一个文件只能有1个公共类（主类）—可以没有主函数
//主函数必须要放在主类中
public class A {
private int a=1;//私有级别
int b=2;//默认级别
protected int c=3;//受保护级别
public int d=4;//公有级别

private void f1(){
    System.out.println("Private"+this.a);
}
void f2(){
    System.out.println("Default"+this.b);
}
protected void f3(){
    System.out.println("Protected"+this.c);
}
public void f4(){
    System.out.println("Public"+this.d);
}

}

class A2{
A aa=new A();
void test(){
System.out.println(aa.a);//a only available in class A
System.out.println(aa.b);
System.out.println(aa.c);
System.out.println(aa.d);
}
}

//final关键字
final class L{
//final修饰成员变量
// 变量本身必须赋值且只能赋值一次，赋值后不可再次更改
final int a=1;//final修饰的是普通的成员变量【定义时赋值】
final int b;
L(){
b=2;//可在构造函数中赋值
}
final static int c=3;//final修饰静态成员变量【定义时赋值】
final static int d;
static {
d=4;//可在静态代码块中赋值
}

//final修饰成员函数
//此函数（f函数）不可被子类重写

class Transport{
int speed;

void run(){
    System.out.println("Transportaion is running");
}

}

class Plane extends Transport{
@Override
void run(){
speed=600;
System.out.println(“Plane is flying at 600 kph on the sky”);
}
}
class Car extends Transport{
@Override
void run(){
speed=120;
System.out.println(“Car is running at 120 kph on the highway”);
}
}
class Boat extends Transport{
@Override
void run(){
speed=30;
System.out.println(“Boat is running at 30 knot on the sky”);
}
}
